def get_group(group_id)
  @@db_groups ||= data_bag('groups')
  @@groups ||= {}
  @@groups[group_id] ||= Mash::new(
    if @@db_groups.include?(group_id)
    then data_bag_item("groups", group_id).to_hash
    else {
        :id => group_id,
        :imaginary => true
      }
    end
    )
end
